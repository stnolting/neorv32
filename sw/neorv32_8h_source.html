253#define __MMREG_BSET(r, m) ((r) |= (m))
254#define __MMREG_BCLR(r, m) ((r) &= ~(m))
255#define __MMREG_BINV(r, m) ((r) ^= (m))
256// 32-bit access
257#ifdef __riscv_a // use atomic RMW instructions (Zaamo)
258#define __MMREG32_BSET(r, m) (neorv32_cpu_amoor( (uint32_t)(&r), (uint32_t)( (m))))
259#define __MMREG32_BCLR(r, m) (neorv32_cpu_amoand((uint32_t)(&r), (uint32_t)(~(m))))
260#define __MMREG32_BINV(r, m) (neorv32_cpu_amoxor((uint32_t)(&r), (uint32_t)( (m))))
261#else // use individual load + modify + write instructions
262#define __MMREG32_BSET(r, m) __MMREG_BSET(r, (uint32_t)m)
263#define __MMREG32_BCLR(r, m) __MMREG_BCLR(r, (uint32_t)m)
264#define __MMREG32_BINV(r, m) __MMREG_BINV(r, (uint32_t)m)
265#endif

268#define __MMREG16_BCLR(r, m) __MMREG_BCLR(r, (uint16_t)m)
