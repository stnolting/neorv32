50inline void __attribute__ ((always_inline)) neorv32_context_save(void) {
51
52 // do not backup x0 and sp
53 asm volatile (
54#ifndef __riscv_32e

160inline void __attribute__ ((always_inline)) neorv32_cpu_store_unsigned_word(uint32_t addr, uint32_t wdata) {
161
162 uint32_t reg_addr = addr;
163 uint32_t reg_data = wdata;
164
165 asm volatile ("sw %[da], 0(%[ad])" : : [da] "r" (reg_data), [ad] "r" (reg_addr));

297/**********************************************************************/
306inline uint32_t __attribute__ ((always_inline)) neorv32_cpu_load_reservate_word(uint32_t addr) {
307
308#if defined __riscv_atomic
309 uint32_t amo_addr = addr;
310 uint32_t amo_rdata;
311
312 asm volatile ("lr.w %[dst], 0(%[addr])" : [dst] "=r" (amo_rdata) : [addr] "r" (amo_addr));
313
314 return amo_rdata;
315#else
316 (void)addr;
317
318 return 0;
319#endif
320}

323/**********************************************************************/
333inline uint32_t __attribute__ ((always_inline)) neorv32_cpu_store_conditional_word(uint32_t addr, uint32_t wdata) {
334
335#if defined __riscv_atomic
336 uint32_t amo_addr = addr;
337 uint32_t amo_wdata = wdata;
338 uint32_t amo_status;
339
340 asm volatile ("sc.w %[dst], %[src], (%[addr])" : [dst] "=r" (amo_status) : [src] "r" (amo_wdata), [addr] "r" (amo_addr));
341
342 return amo_status;
343#else
344 (void)addr;
345 (void)wdata;
346
347 return 1; // always fail
348#endif
349}
